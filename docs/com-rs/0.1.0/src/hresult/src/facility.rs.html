	fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
		if cfg!(features = "strings") {
			if let Some(name) = name(*self) {
				name.fmt(f)
			}
			else {
				write!(f, "Facility({})", self.0)
			}
		}
		else {
			write!(f, "Facility({})", self.0)
		}
	}
}
