Creates a new RefProperty.

Given a name, description, a reference to the underlying value and an appropriate default.

Example

// The underlying data wrapped in a `RefCell`.
use std::cell::RefCell;
let var = RefCell::new(String::from("Hello, World!"));

// The variable wrapped in a `RefProperty`.
use cvar::{RefProperty, IProperty};
let prop = RefProperty::new("prop", "property description", &var, "Hello, Default!");
assert_eq!(prop.get().unwrap(), "Hello, World!");
prop.reset();
assert_eq!(*var.borrow(), "Hello, Default!");
