pub fn name_cmp<'a>(id: &'a str, name: &str) -> NameMatch<'a>

Looks if this name matches the identifier.

Identifier strings are a list of names separated by a . character, eg foo.bar.

# Examples

 assert_eq!(name_cmp("foo", "foo"), NameMatch::Exact);
 assert_eq!(name_cmp("foo.bar", "foo"), NameMatch::Sub("bar"));
 assert_eq!(name_cmp("foo.bar.baz", "foo.bar"), NameMatch::Sub("baz"));
 assert_eq!(name_cmp("fooz", "foo"), NameMatch::Mis);
 assert_eq!(name_cmp("foo.bar.baz", "bar.baz"), NameMatch::Mis);
